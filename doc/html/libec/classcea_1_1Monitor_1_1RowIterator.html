Detailed Description

template<int TAG, typename CAST>
class cea::Monitor::RowIterator< TAG, CAST >

Row "tag" Iterator.

This iterator allow to select and directly cast only certain item follow their tag.
This iterator is optimized and should be used if not all columns/rows of a column/row need to be updated.
If all rows/columns need to be updated then use classic iterator of GridModel::columns or GridModel::rows member.

Example

class MyMonitor : public Monitor
{
// Code for columns, rows ...
void onRowUpdate(Row& r)
{
if (r.tag == FEEDER_PROCESS_ITEM) {
Process& p = cast<Process>(r);
// Update Sensor U64
for(ColumnIterator<SENSOR_U64, Sensor> s = this;
s != 0; ++s)
{
setValue(r, s, s->getValue(p.getPid()).U64);
}
// Update Sensor Float
for(ColumnIterator<SENSOR_FLOAT, Sensor> s = this;
s != 0; ++s)
{
setValue(r, s, s->getValue(p.getPid()).Float);
}
}
}
};
